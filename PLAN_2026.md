```typescript
import type { Tool, ToolEvaluation } from '@projectmodern/types';

export class ProjectModernClient {
  constructor(private baseUrl: string) {}

  async evaluateTool(platform: string, name: string): Promise<ToolEvaluation> {
    const response = await fetch(`${this.baseUrl}/tools/${platform}/${name}`);
    if (!response.ok) throw new Error(`Failed to evaluate ${name}`);
    return response.json();
  }

  async compareTools(tools: Array<{ platform: string; name: string }>): Promise<ComparisonResult> {
    const response = await fetch(`${this.baseUrl}/compare`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ tools }),
    });
    return response.json();
  }

  async semanticSearch(query: string, limit = 10): Promise<SemanticSearchResult> {
    const params = new URLSearchParams({ q: query, limit: String(limit) });
    const response = await fetch(`${this.baseUrl}/search?${params}`);
    return response.json();
  }
}
```

```typescript
import type { Tool, CompositeScore } from '@projectmodern/types';

import { LibrariesIoIntegration } from './integrations/libraries-io.js';

  version: process.env.npm_package_version,
}));

// Evaluate tool
fastify.get<{
  Params: { platform: string; name: string };
}>('/tools/:platform/:name', async (request, reply) => {
  const { platform, name } = request.params;

    LibrariesIoIntegration.getData(platform, name),

    maintenance: calculateMaintenanceScore(librariesData),
    popularity: calculatePopularityScore(githubData),
    ecosystem: calculateEcosystemScore(librariesData),
  });

  return {
    name,
    platform,
    evaluation,
    sources: {
      scorecard: scorecardData,
      libraries: librariesData,
      github: githubData,
    },
  };
```

```typescript
import OpenAI from 'openai';

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

export async function generateEmbedding(text: string): Promise<number[]> {
  const response = await openai.embeddings.create({
    model: 'text-embedding-3-small',  // 1536 dimensions
    input: text,
    encoding_format: 'float',
  });

  return response.data[0].embedding;
}

// Batch embedding for efficiency
export async function generateEmbeddings(texts: string[]): Promise<number[][]> {
  const response = await openai.embeddings.create({
    model: 'text-embedding-3-small',
    input: texts,
  });

  return response.data.map(d => d.embedding);
}
```

```typescript
// services/semantic-search/src/database.ts
import { Pool } from 'pg';

const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
});

export async function upsertTool(tool: ToolWithEmbedding): Promise<void> {
  const query = `
    INSERT INTO tools (name, platform, description, keywords, embedding, score, metadata)
    VALUES ($1, $2, $3, $4, $5, $6, $7)
    ON CONFLICT (name, platform) DO UPDATE SET
      description = EXCLUDED.description,
      keywords = EXCLUDED.keywords,
      embedding = EXCLUDED.embedding,
      score = EXCLUDED.score,
      metadata = EXCLUDED.metadata,
      updated_at = NOW()
  `;

  await pool.query(query, [
    tool.name,
    tool.platform,
    tool.description,
    tool.keywords,
    `[${tool.embedding.join(',')}]`,  // pgvector format
    tool.score,
    JSON.stringify(tool.metadata),
  ]);
}

export async function semanticSearch(
  embedding: number[],
  limit = 10,
  minSimilarity = 0.7
): Promise<SearchResult[]> {
  const query = `
    SELECT 
      name,
      platform,
      description,
      score,
      1 - (embedding <=> $1) as similarity
    FROM tools
    WHERE 1 - (embedding <=> $1) > $2
    ORDER BY embedding <=> $1
    LIMIT $3
  `;

  const result = await pool.query(query, [
    `[${embedding.join(',')}]`,
    minSimilarity,
    limit,
  ]);

  return result.rows;
}
```

```typescript
// services/policy-engine/src/index.ts
import { Policy, ToolEvaluation, PolicyViolation } from '@projectmodern/types';

export class PolicyEngine {
  constructor(private policies: Policy[]) {}

  evaluate(evaluation: ToolEvaluation): PolicyViolation[] {
    const violations: PolicyViolation[] = [];

    for (const policy of this.policies) {
      const violation = this.checkPolicy(policy, evaluation);
      if (violation) violations.push(violation);
    }

    return violations;
  }

  private checkPolicy(policy: Policy, evaluation: ToolEvaluation): PolicyViolation | null {
    switch (policy.type) {
      case 'minScore':
        if (evaluation.composite < policy.value) {
          return {
            policy,
            message: `Score ${evaluation.composite} below minimum ${policy.value}`,
            severity: 'error',
          };
        }
        break;

      case 'blocked':
        // Check if tool is in blocklist
        break;

      case 'license':
        // Check license compatibility
        break;

      case 'maintenance':
        if (evaluation.breakdown.maintenance < policy.value) {
          return {
            policy,
            message: `Maintenance score too low`,
            severity: 'warning',
          };
        }
        break;
    }

    return null;
  }
}
```

#### Week 10: API Keys + Rate Limiting

```typescript
// services/api-gateway/src/rate-limit.ts
import { FastifyPluginAsync } from 'fastify';
import { Redis } from 'ioredis';

const redis = new Redis(process.env.REDIS_URL);

export const rateLimitPlugin: FastifyPluginAsync = async (fastify) => {
  fastify.addHook('onRequest', async (request, reply) => {
    const apiKey = request.headers['x-api-key'] as string;
    if (!apiKey) return;

    const key = `rate_limit:${apiKey}`;
    const limit = 1000; // requests per hour
    const window = 3600; // 1 hour

    const current = await redis.incr(key);
    if (current === 1) {
      await redis.expire(key, window);
    }

    const remaining = Math.max(0, limit - current);
    reply.header('X-RateLimit-Limit', limit);
    reply.header('X-RateLimit-Remaining', remaining);

    if (current > limit) {
      reply.status(429).send({ error: 'Rate limit exceeded' });
      return;
    }
  });
};
```

## Appendix B: Database Schema

```sql
-- Core tables
CREATE EXTENSION IF NOT EXISTS vector;

CREATE TABLE tools (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(255) NOT NULL,
  platform VARCHAR(50) NOT NULL,
  description TEXT,
  keywords TEXT[],
  embedding VECTOR(1536),
  score JSONB NOT NULL,
  metadata JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(name, platform)
);

CREATE INDEX idx_tools_embedding ON tools USING ivfflat (embedding vector_cosine_ops);
CREATE INDEX idx_tools_platform ON tools(platform);

-- Users and teams
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email VARCHAR(255) UNIQUE NOT NULL,
  name VARCHAR(255),
  team_id UUID,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE teams (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(255) NOT NULL,
  slug VARCHAR(255) UNIQUE NOT NULL,
  settings JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Policies
CREATE TABLE policies (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  team_id UUID REFERENCES teams(id),
  name VARCHAR(255) NOT NULL,
  type VARCHAR(50) NOT NULL, -- 'minScore', 'blocked', 'license', etc.
  config JSONB NOT NULL,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Audit log
CREATE TABLE audit_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  timestamp TIMESTAMPTZ DEFAULT NOW(),
  user_id UUID REFERENCES users(id),
  team_id UUID REFERENCES teams(id),
  action VARCHAR(100) NOT NULL,
  resource_type VARCHAR(100),
  resource_id VARCHAR(255),
  metadata JSONB,
  ip_address INET
);

CREATE INDEX idx_audit_user ON audit_events(user_id);
CREATE INDEX idx_audit_team ON audit_events(team_id);
CREATE INDEX idx_audit_timestamp ON audit_events(timestamp);
```
